cmake_minimum_required(VERSION 3.15)
project(chan_minimal LANGUAGES CXX)

# 设置编译选项
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 强制32位编译
if(CMAKE_SIZEOF_VOID_P EQUAL 8)
    message(FATAL_ERROR "必须使用32位编译器! 请使用 cmake -A Win32")
endif()

# 设置输出目录
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

# 创建最小版DLL
add_library(chan_min SHARED
    src/tdx_minimal.cpp
)

# 设置导出定义文件
set_target_properties(chan_min PROPERTIES
    OUTPUT_NAME "chan_min"
    DEFINE_FILE "${CMAKE_SOURCE_DIR}/chan.def"
)

# Windows特定设置
if(MSVC)
    target_compile_options(chan_min PRIVATE /W3 /O2 /MT)
    target_compile_definitions(chan_min PRIVATE WIN32 _WINDOWS _USRDLL NDEBUG)
endif()
