{缠论笔 v6.2 - 主图叠加}
{显示笔端点和连线}

{==== 获取数据 ====}
BI:=TDXDLL1(2, H, L, C);   {笔类型: 1=顶, -1=底, 0=非端点}
KXG:=TDXDLL1(20, H, L, C); {笔高点价格}
KXD:=TDXDLL1(21, H, L, C); {笔低点价格}

{==== 笔连线 ====}
DRAWLINE(BI=1, KXG, BI=-1, KXD, 0), COLORYELLOW;
DRAWLINE(BI=-1, KXD, BI=1, KXG, 0), COLORYELLOW;

{==== 笔端点标记 ====}
DRAWICON(BI=1, KXG, 14);
DRAWICON(BI=-1, KXD, 13);
